// ─── FETCH RISULTATI ──────────────────────────────────────────────────

async function fetchRisultati() {

  const btn = document.getElementById('btnRefresh');

  const info = document.getElementById('statusInfo');

  btn.disabled = true;

  info.innerHTML = '<span class="spinner"></span> Aggiornamento...';

  try {

    const API_KEY = '46e1be88d1f141c9a5392183c875d71c';

    // Aggiunto timestamp anti-cache nell'URL dell'API

    const url = `https://api.football-data.org/v4/competitions/WC/matches?stage=LAST_32&season=2026&nocache=${new Date().getTime()}`;

    const proxy = `https://corsproxy.io/?url=${encodeURIComponent(url)}`;

    const res = await fetch(proxy, {

      headers: { 'X-Auth-Token': API_KEY }

    });

    const data = await res.json();

    if (data.matches) {

      risultati = new Array(16).fill(null);

      scores = new Array(16).fill(null);

      for (const m of data.matches) {

        const hName = m.homeTeam?.name || '';

        const aName = m.awayTeam?.name || '';

        const idx = TEAM_MAP[hName] ?? TEAM_MAP[aName] ?? -1;

        if (idx === -1) continue;

        const status = m.status;

       

        // MODIFICA LIVE CRUCIALE: Estrae i dati corretti anche in tempo reale

        let hScore = m.score?.fullTime?.home;

        let aScore = m.score?.fullTime?.away;

        if ((status === 'IN_PLAY' || status === 'PAUSED') && (hScore === null || hScore === undefined)) {

          hScore = m.score?.regularTime?.home ?? m.score?.halfTime?.home ?? 0;

          aScore = m.score?.regularTime?.away ?? m.score?.halfTime?.away ?? 0;

         

          if (m.score?.home !== undefined && m.score?.home !== null) hScore = m.score.home;

          if (m.score?.away !== undefined && m.score?.away !== null) aScore = m.score.away;

        }

        if ((status === 'FINISHED' || status === 'IN_PLAY' || status === 'PAUSED') && hScore !== null && aScore !== null) {

          scores[idx] = { h: hScore, a: aScore, live: status !== 'FINISHED' };

          if (hScore > aScore) risultati[idx] = '1';

          else if (hScore === aScore) risultati[idx] = 'X';

          else risultati[idx] = '2';

        }

      }

      lastUpdate = new Date();

      info.innerHTML = `✅ <strong>Aggiornato</strong> alle ${lastUpdate.toLocaleTimeString('it-IT')} · ${risultati.filter(r=>r!==null).length}/16 partite in corso o concluse`;

    } else {

      throw new Error('Dati non disponibili');

    }

  } catch(e) {

    info.innerHTML = `⚠️ Errore aggiornamento: ${e.message}`;

    console.error(e);

  }

  renderMatches();

  renderClassifica();

  btn.disabled = false;

}

// ─── RENDER PARTITE ───────────────────────────────────────────────────

function renderMatches() {

  const grid = document.getElementById('matchesGrid');

  grid.innerHTML = '';

  PARTITE_INFO.forEach((p, i) => {

    const s = scores[i];

    const r = risultati[i];

    const div = document.createElement('div');

    div.className = 'match-card';

    let scoreStr = s ? `${s.h} – ${s.a}${s.live ? ' <span style="color:#ef4444;animation:pulse 1s infinite">🔴</span>' : ''}` : '– –';

    let esitoStr = '';

    let esitoClass = 'pending';

    if (r === '1') { esitoStr = '1 (vittoria casa)'; esitoClass = 'e1'; }

    else if (r === 'X') { esitoStr = 'X (pareggio)'; esitoClass = 'eX'; }

    else if (r === '2') { esitoStr = '2 (vittoria ospite)'; esitoClass = 'e2'; }

    else esitoStr = 'Non ancora giocata';

    div.innerHTML = `

      <div class="partita-num">Partita ${i+1}</div>

      <div class="teams">

        <div class="team-name">${p.home}</div>

        <div class="score-box">${scoreStr}</div>

        <div class="team-name right">${p.away}</div>

      </div>

      <div class="esito ${esitoClass}">${esitoStr}</div>`;

    grid.appendChild(div);

  });

}

// ─── RENDER CLASSIFICA ────────────────────────────────────────────────

function renderClassifica() {

  const rows = PARTECIPANTI.map((p, i) => {

    const pron = PRONOSTICI[i];

    let pts16 = 0;

    risultati.forEach((r, m) => {

      if (r && pron[m] && pron[m] === r) pts16++;

    });

    return { ...p, pts16, tot: p.base + pts16, pron };

  });

  rows.sort((a, b) => b.tot - a.tot || b.base - a.base);

  const tbody = document.getElementById('tbody');

  tbody.innerHTML = '';

  rows.forEach((p, rank) => {

    const tr = document.createElement('tr');

    let medal = '';

    if (rank === 0) medal = '🥇';

    else if (rank === 1) medal = '🥈';

    else if (rank === 2) medal = '🥉';

    let cells = '';

    for (let m = 0; m < 16; m++) {

      const v = p.pron[m];

      const r = risultati[m];

      let cls = '';

      if (v === '1') cls = 'v1';

      else if (v === 'X') cls = 'vX';

      else if (v === '2') cls = 'v2';

      else cls = 'empty';

      if (r && v && r === v) cls += ' ok';

      cells += `<td><div class="cell ${cls}">${v || '–'}</div></td>`;

    }

    tr.innerHTML = `

      <td class="rank">${medal || (rank+1)}</td>

      <td class="name-col">${p.nome}</td>

      ${cells}

      <td class="pts-base sep">${p.base}</td>

      <td class="pts-16">${p.pts16}</td>

      <td class="pts-tot sep">${p.tot}</td>`;

    tbody.appendChild(tr);

  });

}
